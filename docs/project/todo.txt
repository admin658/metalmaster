"use client";

import React, { useEffect, useRef, useState } from "react";
import { usePathname } from "next/navigation";

/**
 * Shows a splash overlay on every pathname change.
 * Designed for Next.js App Router.
 */
export default function RouteSplash({
  logoSrc = "/assets/metalmaster-logo.png",
  durationMs = 900,     // keep short for navigation
  fadeMs = 250,
  minIntervalMs = 250,  // prevents double-trigger in dev / fast transitions
}) {
  const pathname = usePathname();
  const [visible, setVisible] = useState(false);
  const [closing, setClosing] = useState(false);
  const lastShownAt = useRef(0);
  const timers = useRef([]);

  useEffect(() => {
    const now = Date.now();
    if (now - lastShownAt.current < minIntervalMs) return;
    lastShownAt.current = now;

    // Show immediately on route change
    setVisible(true);
    setClosing(false);

    // Clear old timers
    timers.current.forEach((t) => clearTimeout(t));
    timers.current = [];

    // Start fade-out, then hide
    const t1 = setTimeout(() => setClosing(true), Math.max(0, durationMs - fadeMs));
    const t2 = setTimeout(() => setVisible(false), durationMs);
    timers.current.push(t1, t2);

    return () => {
      timers.current.forEach((t) => clearTimeout(t));
      timers.current = [];
    };
  }, [pathname, durationMs, fadeMs, minIntervalMs]);

  if (!visible) return null;

  return (
    <div className={`mm-splash ${closing ? "mm-splash--out" : ""}`} role="status" aria-label="Loading">
      <div className="mm-splash__bg" aria-hidden="true" />
      <div className="mm-splash__smoke" aria-hidden="true" />
      <div className="mm-splash__embers" aria-hidden="true" />

      <div className="mm-splash__content">
        <div className="mm-splash__logoWrap">
          <img className="mm-splash__logo" src={logoSrc} alt="Metal Master" draggable="false" />
          <div className="mm-splash__glow" aria-hidden="true" />
        </div>
      </div>

      <style>{css}</style>
    </div>
  );
}
